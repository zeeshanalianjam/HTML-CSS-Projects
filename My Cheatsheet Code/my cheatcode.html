		document.write('Data Types <br><br>');

		// 1.string
		var str = 'hello';
		document.write(str);
		document.write('<br>');
		document.write(typeof str + '<br><br>');

		// 2.number
		var num = 34;
		document.write(num);
		document.write('<br>');
		document.write(typeof num + '<br><br>');

		// 3.boolean
		var bol = true;
		document.write(bol);
		document.write('<br>');
		document.write(typeof bol + '<br><br>');

		// 4.undefined
		var und = undefined;
		document.write(und);
		document.write('<br>');
		document.write(typeof und + '<br><br>');

		// 5.array
		var ary = ['html', 'css', 'javascript'];
		document.write(ary);
		document.write('<br>');
		document.write(typeof ary + '<br><br>');

		// 6.object
		var obj = {
			name: 'zeeshan',
			country: 'pakistan',
			visa: 7890987
		};
		document.write(obj);
		document.write('<br>');
		document.write(typeof obj + '<br><br>');
	</script>
